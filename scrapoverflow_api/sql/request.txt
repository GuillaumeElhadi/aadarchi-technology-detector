SELECT
  t.TagName AS Tag,
  COUNT(*) AS Popularite,
  t.ExcerptPostId AS ExcerptPostId,
  t.WikiPostId AS WikiPostId,
  CONCAT('https://stackoverflow.com/tags/', t.TagName) AS TagLink,
  CONCAT('https://stackoverflow.com/edit-tag-wiki/', t.ExcerptPostId) AS ExcerptLink,
  CONCAT('https://stackoverflow.com/edit-tag-wiki/', t.WikiPostId) AS WikiLink
FROM
  Tags AS t
INNER JOIN
  PostTags AS pt ON pt.TagId = t.Id
GROUP BY
  t.TagName, t.ExcerptPostId, t.WikiPostId
ORDER BY
  Popularite DESC
